```typescript
import axios from 'axios';

const API_BASE_URL = import.meta.env.VITE_API_BASE_URL || 'http://localhost:5000/api/v1';

const api = axios.create({
  baseURL: API_BASE_URL,
  headers: { 'Content-Type': 'application/json' },
});

// Request interceptor to add auth token
api.interceptors.request.use((config) => {
  const token = localStorage.getItem('token');
  if (token) {
    config.headers.Authorization = `Bearer ${token}`;
  }
  return config;
});

// Response interceptor to handle errors globally
api.interceptors.response.use(
  (response) => response,
  (error) => {
    if (error.response?.status === 401) {
      // Unauthorized – redirect to login
      localStorage.removeItem('token');
      window.location.href = '/login';
    }
    return Promise.reject(error);
  }
);

export default api;
```

```typescript
import api from './api';
import { TimeSeries, DataPoint, PaginatedResponse } from '../types';

export const datasetService = {
  // Get all datasets with pagination
  getDatasets: (page = 1, limit = 20, isVisible?: boolean) => {
    const params: any = { page, limit };
    if (isVisible !== undefined) params.isVisible = isVisible;
    return api.get<PaginatedResponse<TimeSeries>>('/admin/datasets', { params });
  },

  // Get single dataset with its data points
  getDataset: (id: string, page = 1, limit = 50) =>
    api.get<{ success: true; data: TimeSeries & { dataPoints: DataPoint[]; totalDataPoints: number } }>(
      `/admin/datasets/${id}`,
      { params: { page, limit } }
    ),

  // Create new dataset
  createDataset: (data: Partial<TimeSeries>) =>
    api.post('/admin/datasets', data),

  // Update dataset metadata
  updateDataset: (id: string, data: Partial<TimeSeries>) =>
    api.put(`/admin/datasets/${id}`, data),

  // Soft delete (hide) dataset
  deleteDataset: (id: string) => api.delete(`/admin/datasets/${id}`),

  // Upload initial XLSX
  uploadInitial: (id: string, file: File) => {
    const formData = new FormData();
    formData.append('file', file);
    return api.post(`/admin/datasets/${id}/upload-initial`, formData, {
      headers: { 'Content-Type': 'multipart/form-data' },
    });
  },

  // Append XLSX
  appendData: (id: string, file: File) => {
    const formData = new FormData();
    formData.append('file', file);
    return api.post(`/admin/datasets/${id}/append`, formData, {
      headers: { 'Content-Type': 'multipart/form-data' },
    });
  },

  // Update a data point
  updateDataPoint: (pointId: string, data: Partial<DataPoint>) =>
    api.put(`/admin/datasets/data-points/${pointId}`, data),

  // Delete a data point
  deleteDataPoint: (pointId: string) => api.delete(`/admin/datasets/data-points/${pointId}`),
};
```

### Step 5.5: Create Auth Context (Simplified)

Create `client/src/contexts/AuthContext.tsx`:

```tsx
import React, { createContext, useState, useContext, useEffect } from 'react';
import api from '../services/api';
import { User } from '../types';

interface AuthContextType {
  user: User | null;
  token: string | null;
  login: (token: string, user: User) => void;
  logout: () => void;
  isLoading: boolean;
}

const AuthContext = createContext<AuthContextType | undefined>(undefined);

export const AuthProvider: React.FC<{ children: React.ReactNode }> = ({ children }) => {
  const [user, setUser] = useState<User | null>(null);
  const [token, setToken] = useState<string | null>(localStorage.getItem('token'));
  const [isLoading, setIsLoading] = useState(true);

  useEffect(() => {
    if (token) {
      // Fetch current user
      api.get('/auth/me')
        .then(res => setUser(res.data.data))
        .catch(() => {
          localStorage.removeItem('token');
          setToken(null);
        })
        .finally(() => setIsLoading(false));
    } else {
      setIsLoading(false);
    }
  }, [token]);

  const login = (newToken: string, userData: User) => {
    localStorage.setItem('token', newToken);
    setToken(newToken);
    setUser(userData);
  };

  const logout = () => {
    localStorage.removeItem('token');
    setToken(null);
    setUser(null);
  };

  return (
    <AuthContext.Provider value={{ user, token, login, logout, isLoading }}>
      {children}
    </AuthContext.Provider>
  );
};

export const useAuth = () => {
  const context = useContext(AuthContext);
  if (!context) throw new Error('useAuth must be used within AuthProvider');
  return context;
};
```

```tsx
interface Props {
  children: JSX.Element;
}

const ProtectedRoute: React.FC<Props> = ({ children }) => {
  const { token, isLoading } = useAuth();

  if (isLoading) {
    return <div className="flex items-center justify-center h-screen">Loading...</div>;
  }

  if (!token) {
    return <Navigate to="/login" replace />;
  }

  return children;
};
```

`client/src/pages/admin/DataDashboard.tsx`:

```tsx
import React, { useEffect, useState } from 'react';
import { Link } from 'react-router-dom';
import { datasetService } from '../../services/dataset.service';
import { TimeSeries } from '../../types';
import toast from 'react-hot-toast';
import { PencilIcon, TrashIcon, EyeIcon, EyeSlashIcon } from '@heroicons/react/24/outline';

const DataDashboard: React.FC = () => {
  const [datasets, setDatasets] = useState<TimeSeries[]>([]);
  const [page, setPage] = useState(1);
  const [totalPages, setTotalPages] = useState(1);
  const [loading, setLoading] = useState(false);

  const fetchDatasets = async () => {
    setLoading(true);
    try {
      const res = await datasetService.getDatasets(page, 20);
      setDatasets(res.data.data);
      setTotalPages(res.data.pagination.pages);
    } catch (error) {
      toast.error('Failed to load datasets');
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    fetchDatasets();
  }, [page]);

  const handleDelete = async (id: string) => {
    if (!confirm('Are you sure you want to hide this dataset? It will no longer be visible to users.')) return;
    try {
      await datasetService.deleteDataset(id);
      toast.success('Dataset hidden');
      fetchDatasets();
    } catch (error) {
      toast.error('Failed to hide dataset');
    }
  };

  return (
    <div>
      <div className="flex justify-between items-center mb-6">
        <h1 className="text-2xl font-bold text-[#723E31]">Datasets</h1>
        <Link
          to="/admin/data/new"
          className="bg-[#A67564] text-white px-4 py-2 rounded hover:bg-[#723E31]"
        >
          + New Dataset
        </Link>
      </div>

      {loading && <div>Loading...</div>}

      <div className="bg-white shadow overflow-hidden sm:rounded-md">
        <ul className="divide-y divide-gray-200">
          {datasets.map((ds) => (
            <li key={ds.id} className="px-6 py-4 hover:bg-gray-50">
              <div className="flex items-center justify-between">
                <div className="flex-1">
                  <Link to={`/admin/data/${ds.id}`} className="block">
                    <p className="text-lg font-medium text-[#723E31]">{ds.name}</p>
                    <p className="text-sm text-gray-500">
                      Unit: {ds.defaultUnit} | Price: ₹{ds.priceINR} | Data points: {ds._count?.dataPoints || 0}
                    </p>
                    {ds.description && <p className="text-sm text-gray-600 mt-1">{ds.description}</p>}
                  </Link>
                </div>
                <div className="flex space-x-2">
                  {ds.isVisible ? (
                    <EyeIcon className="h-5 w-5 text-green-600" title="Visible" />
                  ) : (
                    <EyeSlashIcon className="h-5 w-5 text-gray-400" title="Hidden" />
                  )}
                  <Link to={`/admin/data/${ds.id}/edit`} className="text-[#A67564] hover:text-[#723E31]">
                    <PencilIcon className="h-5 w-5" />
                  </Link>
                  <button onClick={() => handleDelete(ds.id)} className="text-red-600 hover:text-red-800">
                    <TrashIcon className="h-5 w-5" />
                  </button>
                </div>
              </div>
            </li>
          ))}
        </ul>
      </div>

      {/* Pagination */}
      {totalPages > 1 && (
        <div className="flex justify-center mt-6 space-x-2">
          <button
            onClick={() => setPage(p => Math.max(1, p-1))}
            disabled={page === 1}
            className="px-3 py-1 border rounded disabled:opacity-50"
          >
            Previous
          </button>
          <span className="px-3 py-1">Page {page} of {totalPages}</span>
          <button
            onClick={() => setPage(p => Math.min(totalPages, p+1))}
            disabled={page === totalPages}
            className="px-3 py-1 border rounded disabled:opacity-50"
          >
            Next
          </button>
        </div>
      )}
    </div>
  );
};

export default DataDashboard;
```

```tsx
import React, { useEffect, useState } from 'react';
import { useForm } from 'react-hook-form';
import { useNavigate, useParams } from 'react-router-dom';
import { datasetService } from '../../services/dataset.service';
import toast from 'react-hot-toast';

interface FormData {
  name: string;
  description?: string;
  defaultUnit: string;
  priceINR: number;
  metadata?: string;
}

const DatasetForm: React.FC = () => {
  const { id } = useParams<{ id: string }>();
  const navigate = useNavigate();
  const [loading, setLoading] = useState(false);
  const { register, handleSubmit, setValue, formState: { errors } } = useForm<FormData>();

  useEffect(() => {
    if (id) {
      // Edit mode: fetch dataset data
      setLoading(true);
      datasetService.getDataset(id, 1, 1) // we just need metadata
        .then(res => {
          const ds = res.data.data;
          setValue('name', ds.name);
          setValue('description', ds.description || '');
          setValue('defaultUnit', ds.defaultUnit);
          setValue('priceINR', ds.priceINR);
          setValue('metadata', ds.metadata ? JSON.stringify(ds.metadata, null, 2) : '');
        })
        .catch(() => toast.error('Failed to load dataset'))
        .finally(() => setLoading(false));
    }
  }, [id, setValue]);

  const onSubmit = async (data: FormData) => {
    try {
      const payload = {
        ...data,
        priceINR: Number(data.priceINR),
        metadata: data.metadata ? JSON.parse(data.metadata) : undefined,
      };
      if (id) {
        await datasetService.updateDataset(id, payload);
        toast.success('Dataset updated');
      } else {
        await datasetService.createDataset(payload);
        toast.success('Dataset created');
      }
      navigate('/admin/data');
    } catch (error: any) {
      toast.error(error.response?.data?.error || 'Operation failed');
    }
  };

  if (loading) return <div>Loading...</div>;

  return (
    <div className="max-w-2xl mx-auto bg-white p-6 rounded shadow">
      <h1 className="text-2xl font-bold text-[#723E31] mb-6">
        {id ? 'Edit Dataset' : 'Create New Dataset'}
      </h1>
      <form onSubmit={handleSubmit(onSubmit)} className="space-y-4">
        <div>
          <label className="block text-sm font-medium mb-1">Name *</label>
          <input
            {...register('name', { required: 'Name is required' })}
            className="w-full border rounded px-3 py-2"
          />
          {errors.name && <p className="text-red-500 text-sm mt-1">{errors.name.message}</p>}
        </div>

        <div>
          <label className="block text-sm font-medium mb-1">Description</label>
          <textarea
            {...register('description')}
            rows={3}
            className="w-full border rounded px-3 py-2"
          />
        </div>

        <div>
          <label className="block text-sm font-medium mb-1">Default Unit *</label>
          <input
            {...register('defaultUnit', { required: 'Default unit is required' })}
            className="w-full border rounded px-3 py-2"
            placeholder="e.g., barrel, kWh"
          />
          {errors.defaultUnit && <p className="text-red-500 text-sm mt-1">{errors.defaultUnit.message}</p>}
        </div>

        <div>
          <label className="block text-sm font-medium mb-1">Price (INR) *</label>
          <input
            type="number"
            step="0.01"
            {...register('priceINR', { required: 'Price is required', min: 0 })}
            className="w-full border rounded px-3 py-2"
          />
          {errors.priceINR && <p className="text-red-500 text-sm mt-1">{errors.priceINR.message}</p>}
        </div>

        <div>
          <label className="block text-sm font-medium mb-1">Metadata (JSON)</label>
          <textarea
            {...register('metadata')}
            rows={5}
            className="w-full border rounded px-3 py-2 font-mono text-sm"
            placeholder='{"source": "EIA"}'
          />
        </div>

        <div className="flex justify-end space-x-3">
          <button
            type="button"
            onClick={() => navigate('/admin/data')}
            className="px-4 py-2 border rounded hover:bg-gray-100"
          >
            Cancel
          </button>
```

### Step 5.11: Create Dataset Detail Page

`client/src/pages/admin/DatasetDetail.tsx`:

```tsx
import React, { useEffect, useState } from 'react';
import { useParams, Link, useNavigate } from 'react-router-dom';
import { datasetService } from '../../services/dataset.service';
import { TimeSeries, DataPoint } from '../../types';
import toast from 'react-hot-toast';
import { PencilIcon, TrashIcon, ArrowUpTrayIcon } from '@heroicons/react/24/outline';
import DataPointRow from '../../components/DataPointRow';
import UploadModal from '../../components/UploadModal';

const DatasetDetail: React.FC = () => {
  const { id } = useParams<{ id: string }>();
  const navigate = useNavigate();
  const [dataset, setDataset] = useState<TimeSeries | null>(null);
  const [dataPoints, setDataPoints] = useState<DataPoint[]>([]);
  const [page, setPage] = useState(1);
  const [totalPoints, setTotalPoints] = useState(0);
  const [limit] = useState(20);
  const [loading, setLoading] = useState(false);
  const [uploadModalOpen, setUploadModalOpen] = useState<'initial' | 'append' | null>(null);

  const fetchDataset = async () => {
    if (!id) return;
    setLoading(true);
    try {
      const res = await datasetService.getDataset(id, page, limit);
      setDataset(res.data.data);
      setDataPoints(res.data.data.dataPoints);
      setTotalPoints(res.data.data.totalDataPoints);
    } catch (error) {
      toast.error('Failed to load dataset');
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    fetchDataset();
  }, [id, page]);

  const handleDeletePoint = async (pointId: string) => {
    if (!confirm('Delete this data point?')) return;
    try {
      await datasetService.deleteDataPoint(pointId);
      toast.success('Data point deleted');
      fetchDataset();
    } catch (error) {
      toast.error('Failed to delete');
    }
  };

  const handleUpdatePoint = async (pointId: string, data: Partial<DataPoint>) => {
    try {
      await datasetService.updateDataPoint(pointId, data);
      toast.success('Updated');
      fetchDataset();
    } catch (error) {
      toast.error('Update failed');
    }
  };

  const handleUpload = async (file: File, type: 'initial' | 'append') => {
    if (!id) return;
    try {
      if (type === 'initial') {
        await datasetService.uploadInitial(id, file);
        toast.success('Initial data uploaded');
      } else {
        await datasetService.appendData(id, file);
        toast.success('Data appended');
      }
      setUploadModalOpen(null);
      fetchDataset();
    } catch (error: any) {
      toast.error(error.response?.data?.error || 'Upload failed');
    }
  };

  if (loading && !dataset) return <div>Loading...</div>;
  if (!dataset) return <div>Dataset not found</div>;

  const totalPages = Math.ceil(totalPoints / limit);

  return (
    <div>
      <div className="mb-4 flex items-center justify-between">
        <h1 className="text-2xl font-bold text-[#723E31]">{dataset.name}</h1>
        <div className="space-x-2">
          <button
            onClick={() => setUploadModalOpen('initial')}
            className="bg-green-600 text-white px-3 py-1 rounded text-sm hover:bg-green-700"
          >
            Upload Initial
          </button>
          <button
            onClick={() => setUploadModalOpen('append')}
            className="bg-blue-600 text-white px-3 py-1 rounded text-sm hover:bg-blue-700"
          >
            Append Data
          </button>
          <Link
            to={`/admin/data/${id}/edit`}
            className="bg-[#A67564] text-white px-3 py-1 rounded text-sm hover:bg-[#723E31]"
          >
            Edit Metadata
          </Link>
        </div>
      </div>

      <div className="bg-white p-4 rounded shadow mb-6">
        <p><span className="font-medium">Description:</span> {dataset.description || 'N/A'}</p>
        <p><span className="font-medium">Default Unit:</span> {dataset.defaultUnit}</p>
        <p><span className="font-medium">Price:</span> ₹{dataset.priceINR}</p>
        <p><span className="font-medium">Visibility:</span> {dataset.isVisible ? 'Visible' : 'Hidden'}</p>
        {dataset.metadata && (
          <pre className="bg-gray-100 p-2 rounded mt-2 text-xs">{JSON.stringify(dataset.metadata, null, 2)}</pre>
        )}
      </div>

      <h2 className="text-xl font-semibold mb-4">Data Points</h2>

      {dataPoints.length === 0 ? (
        <p className="text-gray-500">No data points yet. Upload an XLSX file.</p>
      ) : (
        <>
          <div className="bg-white shadow overflow-hidden sm:rounded-md">
            <table className="min-w-full divide-y divide-gray-200">
              <thead className="bg-gray-50">
                <tr>
                  <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">Date</th>
                  <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">Value</th>
                  <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">Unit Override</th>
                  <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">Note</th>
                  <th className="px-6 py-3 text-right text-xs font-medium text-gray-500 uppercase tracking-wider">Actions</th>
                </tr>
              </thead>
              <tbody className="bg-white divide-y divide-gray-200">
                {dataPoints.map((point) => (
                  <DataPointRow
                    key={point.id}
                    point={point}
                    onDelete={handleDeletePoint}
                    onUpdate={handleUpdatePoint}
                  />
                ))}
              </tbody>
            </table>
          </div>

          {/* Pagination */}
          {totalPages > 1 && (
            <div className="flex justify-center mt-4 space-x-2">
              <button
                onClick={() => setPage(p => Math.max(1, p-1))}
                disabled={page === 1}
                className="px-3 py-1 border rounded disabled:opacity-50"
              >
                Previous
              </button>
              <span className="px-3 py-1">Page {page} of {totalPages}</span>
              <button
                onClick={() => setPage(p => Math.min(totalPages, p+1))}
                disabled={page === totalPages}
                className="px-3 py-1 border rounded disabled:opacity-50"
              >
                Next
              </button>
            </div>
          )}
        </>
      )}

      {/* Upload Modal */}
      <UploadModal
        isOpen={uploadModalOpen !== null}
        onClose={() => setUploadModalOpen(null)}
        onUpload={(file) => handleUpload(file, uploadModalOpen!)}
        title={uploadModalOpen === 'initial' ? 'Upload Initial XLSX' : 'Append XLSX'}
      />
    </div>
  );
};

export default DatasetDetail;
```

### Step 5.12: Create DataPointRow Component (Inline Edit)

`client/src/components/DataPointRow.tsx`:

```tsx
import React, { useState } from 'react';
import { DataPoint } from '../types';
import { PencilIcon, TrashIcon, CheckIcon, XMarkIcon } from '@heroicons/react/24/outline';

interface Props {
  point: DataPoint;
  onDelete: (id: string) => void;
  onUpdate: (id: string, data: Partial<DataPoint>) => Promise<void>;
}

const DataPointRow: React.FC<Props> = ({ point, onDelete, onUpdate }) => {
  const [editing, setEditing] = useState(false);
  const [formData, setFormData] = useState({
    value: point.value,
    unitOverride: point.unitOverride || '',
    note: point.note || '',
    date: point.date.slice(0, 10), // YYYY-MM-DD
  });

  const handleSave = async () => {
    await onUpdate(point.id, {
      value: formData.value,
      unitOverride: formData.unitOverride || null,
      note: formData.note || null,
      date: new Date(formData.date).toISOString(),
    });
    setEditing(false);
  };

  if (editing) {
    return (
      <tr>
        <td className="px-6 py-4">
          <input
            type="date"
            value={formData.date}
            onChange={(e) => setFormData({ ...formData, date: e.target.value })}
            className="border rounded px-2 py-1 w-full"
          />
        </td>
        <td className="px-6 py-4">
          <input
            type="number"
            step="any"
            value={formData.value}
            onChange={(e) => setFormData({ ...formData, value: parseFloat(e.target.value) })}
            className="border rounded px-2 py-1 w-full"
          />
        </td>
        <td className="px-6 py-4">
          <input
            type="text"
            value={formData.unitOverride}
            onChange={(e) => setFormData({ ...formData, unitOverride: e.target.value })}
            className="border rounded px-2 py-1 w-full"
            placeholder="Unit"
          />
        </td>
        <td className="px-6 py-4">
          <input
            type="text"
            value={formData.note}
            onChange={(e) => setFormData({ ...formData, note: e.target.value })}
            className="border rounded px-2 py-1 w-full"
            placeholder="Note"
          />
        </td>
        <td className="px-6 py-4 text-right space-x-2">
          <button onClick={handleSave} className="text-green-600 hover:text-green-800">
            <CheckIcon className="h-5 w-5 inline" />
          </button>
          <button onClick={() => setEditing(false)} className="text-gray-600 hover:text-gray-800">
            <XMarkIcon className="h-5 w-5 inline" />
          </button>
        </td>
      </tr>
    );
  }

  return (
    <tr>
      <td className="px-6 py-4">{new Date(point.date).toLocaleDateString()}</td>
      <td className="px-6 py-4">{point.value}</td>
      <td className="px-6 py-4">{point.unitOverride || '-'}</td>
      <td className="px-6 py-4">{point.note || '-'}</td>
      <td className="px-6 py-4 text-right space-x-2">
        <button onClick={() => setEditing(true)} className="text-[#A67564] hover:text-[#723E31]">
          <PencilIcon className="h-5 w-5 inline" />
        </button>
        <button onClick={() => onDelete(point.id)} className="text-red-600 hover:text-red-800">
          <TrashIcon className="h-5 w-5 inline" />
        </button>
      </td>
    </tr>
  );
};

export default DataPointRow;
```

### Step 5.13: Create UploadModal Component

`client/src/components/UploadModal.tsx`:

```tsx
import React, { useState } from 'react';
import { Dialog } from '@headlessui/react';
import { XMarkIcon } from '@heroicons/react/24/outline';

interface Props {
  isOpen: boolean;
  onClose: () => void;
  onUpload: (file: File) => void;
  title: string;
}

const UploadModal: React.FC<Props> = ({ isOpen, onClose, onUpload, title }) => {
  const [file, setFile] = useState<File | null>(null);
  const [uploading, setUploading] = useState(false);

  const handleFileChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    if (e.target.files && e.target.files[0]) {
      setFile(e.target.files[0]);
    }
  };

  const handleSubmit = async () => {
    if (!file) return;
    setUploading(true);
    try {
      await onUpload(file);
      setFile(null);
    } finally {
      setUploading(false);
    }
  };

  return (
    <Dialog open={isOpen} onClose={onClose} className="relative z-50">
      <div className="fixed inset-0 bg-black/30" aria-hidden="true" />
      <div className="fixed inset-0 flex items-center justify-center p-4">
        <Dialog.Panel className="mx-auto max-w-md rounded bg-white p-6">
          <div className="flex justify-between items-center mb-4">
            <Dialog.Title className="text-lg font-medium">{title}</Dialog.Title>
            <button onClick={onClose} className="text-gray-400 hover:text-gray-600">
              <XMarkIcon className="h-5 w-5" />
            </button>
          </div>

          <input
            type="file"
            accept=".xlsx, .xls"
            onChange={handleFileChange}
            className="mb-4 block w-full text-sm text-gray-500 file:mr-4 file:py-2 file:px-4 file:rounded file:border-0 file:text-sm file:font-semibold file:bg-[#F2F2F2] file:text-[#723E31] hover:file:bg-[#D9B18E]"
          />

          <div className="flex justify-end space-x-2">
            <button
              onClick={onClose}
              className="px-4 py-2 border rounded hover:bg-gray-100"
              disabled={uploading}
            >
              Cancel
            </button>
            <button
              onClick={handleSubmit}
              disabled={!file || uploading}
              className="px-4 py-2 bg-[#A67564] text-white rounded hover:bg-[#723E31] disabled:opacity-50"
            >
              {uploading ? 'Uploading...' : 'Upload'}
            </button>
          </div>
        </Dialog.Panel>
      </div>
    </Dialog>
  );
};

export default UploadModal;
```
