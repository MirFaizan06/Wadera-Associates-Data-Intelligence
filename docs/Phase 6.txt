```typescript
import { Request, Response } from 'express';
import { PrismaClientílient } from '@prisma/client';

// Get all visible datasets (public listing)
export const getPublicDatasets = async (req: Request, res: Response) => {
  try {
    const page = parseInt(req.query.page as string) || 1;
    const limit = parseInt(req.query.limit as string) || 20;
    const skip = (page - 1) * limit;
    const search = req.query.search as string || '';

    const where: any = { isVisible: true };
    if (search) {
      where.name = { contains: search };
    }

    const datasets = await prisma.timeSeries.findMany({
      where,
      select: {
        id: true,
        name: true,
        description: true,
        defaultUnit: true,
        priceINR: true,
        metadata: true,
        _count: { select: { dataPoints: true } },
        // Optionally include a preview of the last few data points (but not full)
        dataPoints: {
          orderBy: { date: 'desc' },
          take: 6, // show last 6 months preview
        },
      },
      skip,
      take: limit,
      orderBy: { createdAt: 'desc' },
    });

    const total = await prisma.timeSeries.count({ where });

    res.json({
      success: true,
      data: datasets,
      pagination: { page, limit, total, pages: Math.ceil(total / limit) },
    });
  } catch (error) {
    console.error(error);
    res.status(500).json({ success: false, error: 'Server error' });
  }
};
```

```typescript
import { Request, Response } from 'express';
import Razorpay from 'razorpay';
import { PrismaClient } from '@prisma/client';
import crypto from 'crypto';

const prisma = new PrismaClient();

const razorpay = new Razorpay({
  key_id: process.env.RAZORPAY_KEY_ID!,
  key_secret: process.env.RAZORPAY_KEY_SECRET!,
});

// Create Razorpay order
export const createOrder = async (req: Request, res: Response) => {
  try {
    const { datasetId, guestEmail } = req.body;
    const userId = (req as any).user?.id; // may be undefined for guest

    if (!datasetId) {
      return res.status(400).json({ success: false, error: 'Dataset ID required' });
    }

    const dataset = await prisma.timeSeries.findUnique({ where: { id: datasetId } });
    if (!dataset) {
      return res.status(404).json({ success: false, error: 'Dataset not found' });
    }

    const amount = dataset.priceINR * 100; // Razorpay expects paise

    const options = {
      amount,
      currency: 'INR',
      receipt: `receipt_${Date.now()}`,
      notes: {
        datasetId,
        userId: userId || 'guest',
        guestEmail: guestEmail || '',
      },
    };

    const order = await razorpay.orders.create(options);

    // Create purchase record with status PENDING
    await prisma.purchase.create({
      data: {
        userId: userId || null,
        guestEmail: guestEmail || null,
        timeSeriesId: datasetId,
        amountINR: dataset.priceINR,
        razorpayOrderId: order.id,
        status: 'PENDING',
      },
    });

    res.json({
      success: true,
      orderId: order.id,
      amount: order.amount,
      currency: order.currency,
      keyId: process.env.RAZORPAY_KEY_ID,
    });
  } catch (error) {
    console.error(error);
    res.status(500).json({ success: false, error: 'Payment order creation failed' });
  }
};
```

```tsx
import { useEffect, useState } from 'react';
import { Link } from 'react-router-dom';
import api from '../services/api';
import { TimeSeries } from '../types';
import toast from 'react-hot-toast';

export default function DatasetsPage() {
  const [datasets, setDatasets] = useState<TimeSeries[]>([]);
  const [page, setPage] = useState(1);
  const [totalPages, setTotalPages] = useState(1);
  const [loading, setLoading] = useState(false);
  const [search, setSearch] = useState('');

  const fetchDatasets = async () => {
    setLoading(true);
    try {
      const res = await api.get('/public/datasets', { params: { page, limit: 12, search } });
      setDatasets(res.data.data);
      setTotalPages(res.data.pagination.pages);
    } catch (error) {
      toast.error('Failed to load datasets');
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    fetchDatasets();
  }, [page, search]);

  return (
    <div className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8 py-12">
      <h1 className="text-3xl font-bold text-[#723E31] mb-8">Datasets</h1>

      <div className="mb-6">
        <input
          type="text"
          placeholder="Search datasets..."
          value={search}
          onChange={(e) => { setSearch(e.target.value); setPage(1); }}
          className="w-full max-w-md border rounded px-3 py-2"
        />
      </div>

      {loading && <div>Loading...</div>}

      <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6">
        {datasets.map(ds => (
          <Link key={ds.id} to={`/datasets/${ds.id}`} className="block border rounded-lg p-6 hover:shadow-lg transition">
            <h2 className="text-xl font-semibold text-[#723E31] mb-2">{ds.name}</h2>
            <p className="text-gray-600 mb-2 line-clamp-2">{ds.description}</p>
            <p className="text-sm text-gray-500">Unit: {ds.defaultUnit}</p>
            <p className="text-lg font-bold mt-4">₹{ds.priceINR}</p>
            <p className="text-xs text-gray-400">{ds._count?.dataPoints} data points</p>
          </Link>
        ))}
      </div>

      {totalPages > 1 && (
        <div className="flex justify-center mt-8 space-x-2">
          <button
            onClick={() => setPage(p => Math.max(1, p-1))}
            disabled={page === 1}
            className="px-3 py-1 border rounded disabled:opacity-50"
          >
            Previous
          </button>
          <span className="px-3 py-1">Page {page} of {totalPages}</span>
          <button
            onClick={() => setPage(p => Math.min(totalPages, p+1))}
            disabled={page === totalPages}
            className="px-3 py-1 border rounded disabled:opacity-50"
          >
            Next
          </button>
        </div>
      )}
    </div>
  );
}
```

```tsx
import { useEffect, useState } from 'react';
import { useParams, useNavigate } from 'react-router-dom';
import api from '../services/api';
import { useAuth } from '../contexts/AuthContext';
import { LineChart, Line, XAxis, YAxis, CartesianGrid, Tooltip, ResponsiveContainer } from 'recharts';
import toast from 'react-hot-toast';

interface DatasetDetail {
  id: string;
  name: string;
  description: string;
  defaultUnit: string;
  priceINR: number;
  dataPoints: Array<{ date: string; value: number }>;
  hasAccess: boolean;
}

export default function DatasetDetailPage() {
  const { id } = useParams<{ id: string }>();
  const { user } = useAuth();
  const navigate = useNavigate();
  const [dataset, setDataset] = useState<DatasetDetail | null>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    if (!id) return;
    api.get(`/public/datasets/${id}`)
      .then(res => setDataset(res.data.data))
      .catch(() => toast.error('Failed to load dataset'))
      .finally(() => setLoading(false));
  }, [id]);

  const handleBuy = () => {
    if (!user) {
      // Redirect to login with return URL
      navigate(`/login?redirect=/checkout/${id}`);
    } else {
      navigate(`/checkout/${id}`);
    }
  };

  if (loading) return <div className="text-center py-12">Loading...</div>;
  if (!dataset) return <div className="text-center py-12">Dataset not found</div>;

  // Prepare chart data: format dates
  const chartData = dataset.dataPoints.map(dp => ({
    date: new Date(dp.date).toLocaleDateString('en-US', { month: 'short', year: 'numeric' }),
    value: dp.value,
  })).reverse(); // show chronological order

  return (
    <div className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8 py-12">
      <div className="bg-white rounded-lg shadow-lg p-8">
        <h1 className="text-3xl font-bold text-[#723E31] mb-4">{dataset.name}</h1>
        <p className="text-gray-600 mb-6">{dataset.description}</p>

        <div className="grid grid-cols-1 md:grid-cols-2 gap-8 mb-8">
          <div>
            <p><span className="font-medium">Default Unit:</span> {dataset.defaultUnit}</p>
            <p><span className="font-medium">Price:</span> ₹{dataset.priceINR}</p>
            <p><span className="font-medium">Data Points:</span> {dataset.dataPoints.length}</p>
          </div>
          <div className="text-right">
            <button
              onClick={handleBuy}
              className="bg-[#D9B061] text-[#723E31] px-6 py-3 rounded-lg font-semibold hover:bg-[#D9B18E] transition"
            >
              Buy Dataset

          <h2 className="text-xl font-semibold mb-4">Preview</h2>
```

```tsx
  const handlePayment = async () => {
    if (!dataset) return;

    setLoading(true);
    try {
      // Create order
      const orderRes = await api.post('/payment/create-order', {
        datasetId,
        guestEmail: user ? undefined : guestEmail,
      });

      const { orderId, amount, currency, keyId } = orderRes.data;

      const options = {
        key: keyId,
        amount: amount,
        currency: currency,
        name: 'DataMarket',
        description: dataset.name,
        order_id: orderId,
        handler: async (response: any) => {
          // Verify payment
          const verifyRes = await api.post('/payment/verify', {
            razorpay_order_id: response.razorpay_order_id,
            razorpay_payment_id: response.razorpay_payment_id,
            razorpay_signature: response.razorpay_signature,
          });
          if (verifyRes.data.success) {
            toast.success('Payment successful! Check your email for download links.');
            navigate('/payment-success');
          } else {
            toast.error('Payment verification failed');
          }
        },
        prefill: {
          email: user?.email || guestEmail,
        },
        theme: {
          color: '#723E31',
        },
      };

      const rzp = new window.Razorpay(options);
      rzp.open();
    } catch (error: any) {
      toast.error(error.response?.data?.error || 'Payment initiation failed');
    } finally {
      setLoading(false);
    }
  };

  if (!dataset) return <div>Loading...</div>;

  return (
    <div className="max-w-2xl mx-auto px-4 py-12">
      <h1 className="text-2xl font-bold text-[#723E31] mb-6">Checkout</h1>
      <div className="bg-white p-6 rounded shadow">
        <h2 className="text-xl mb-4">{dataset.name}</h2>
        <p className="mb-4">Price: ₹{dataset.priceINR}</p>

        {!user && (
          <div className="mb-4">
            <label className="block text-sm font-medium mb-1">Email for download access</label>
            <input
              type="email"
              value={guestEmail}
              onChange={(e) => setGuestEmail(e.target.value)}
              className="w-full border rounded px-3 py-2"
              required
            />
          </div>
        )}

        <div className="mb-4">
          <label className="flex items-center">
            <input type="checkbox" checked={agree} onChange={(e) => setAgree(e.target.checked)} className="mr-2" />
            <span>I agree to the Terms & Conditions</span>
          </label>
        </div>

        <button
          onClick={handlePayment}
          disabled={loading || !agree || (!user && !guestEmail)}
          className="w-full bg-[#D9B061] text-[#723E31] py-2 rounded font-semibold hover:bg-[#D9B18E] disabled:opacity-50"
        >
          {loading ? 'Processing...' : 'Pay with Razorpay'}
        </button>
      </div>
    </div>
  );
}
```

```typescript
// Generate download token for a dataset (for authenticated users)
export const generateDownloadToken = async (req: Request, res: Response) => {
  try {
    const { datasetId } = req.params;
    const userId = (req as any).user.id;

    // Check if user has license for this dataset
    const license = await prisma.licenseAssignment.findFirst({
      where: {
        userId,
        isActive: true,
        validFrom: { lte: new Date() },
        validTo: { gte: new Date() },
        OR: [
          { datasetIds: { equals: [] } }, // applies to all
          { datasetIds: { array_contains: datasetId } }, // MySQL JSON contains? Need proper query. For simplicity, we'll just check any license.
        ],
      },
    });

    if (!license) {
      return res.status(403).json({ success: false, error: 'No valid license for this dataset' });
    }

    // Generate short-lived token (JWT) for download
    const token = jwt.sign(
      { userId, datasetId, type: 'download' },
      process.env.JWT_SECRET!,
      { expiresIn: '1h' }
    );

    res.json({ success: true, token });
  } catch (error) {
    console.error(error);
    res.status(500).json({ success: false, error: 'Server error' });
  }
};
```

### Step 6.12: Download Endpoint (Backend)

Create `server/src/controllers/download.controller.ts`:

```typescript
import { Request, Response } from 'express';
import jwt from 'jsonwebtoken';
import { PrismaClient } from '@prisma/client';
import ExcelJS from 'exceljs';
import { PassThrough } from 'stream';

const prisma = new PrismaClient();

export const downloadDataset = async (req: Request, res: Response) => {
  try {
    const { datasetId } = req.params;
    const { token } = req.query;

    if (!token) {
      return res.status(401).json({ success: false, error: 'Missing token' });
    }

    // Verify token
    let decoded: any;
    try {
      decoded = jwt.verify(token as string, process.env.JWT_SECRET!);
    } catch (err) {
      return res.status(401).json({ success: false, error: 'Invalid or expired token' });
    }

    // Check if token is for this dataset
    if (decoded.datasetId !== datasetId) {
      return res.status(403).json({ success: false, error: 'Token not valid for this dataset' });
    }

    // Fetch dataset and data points
    const dataset = await prisma.timeSeries.findUnique({
      where: { id: datasetId },
      include: { dataPoints: { orderBy: { date: 'asc' } } },
    });

    if (!dataset) {
      return res.status(404).json({ success: false, error: 'Dataset not found' });
    }

    // Generate Excel file
    const workbook = new ExcelJS.Workbook();
    const worksheet = workbook.addWorksheet('Data');

    worksheet.columns = [
      { header: 'Date', key: 'date', width: 15 },
      { header: 'Value', key: 'value', width: 15 },
      { header: 'Unit', key: 'unit', width: 10 },
      { header: 'Note', key: 'note', width: 30 },
    ];

    dataset.dataPoints.forEach(dp => {
      worksheet.addRow({
        date: dp.date.toISOString().split('T')[0],
        value: dp.value,
        unit: dp.unitOverride || dataset.defaultUnit,
        note: dp.note || '',
      });
    });

    res.setHeader('Content-Type', 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet');
    res.setHeader('Content-Disposition', `attachment; filename="${dataset.name}_data.xlsx"`);

    await workbook.xlsx.write(res);
    res.end();

    // Log download
    await prisma.downloadLog.create({
      data: {
        userId: decoded.userId || null,
        guestEmail: decoded.email || null,
        timeSeriesId: datasetId,
        format: 'XLSX',
        ipAddress: req.ip,
      },
    });
  } catch (error) {
    console.error(error);
    res.status(500).json({ success: false, error: 'Download failed' });
  }
};
```
