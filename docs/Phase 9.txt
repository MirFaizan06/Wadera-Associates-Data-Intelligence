## Phase 9: CMS Manager Dashboard

Note: The code written in this Phase file is only dummy and placeholder. You write your own logic, and especially styles and structure. we are using custom css, not tailwind.

In this phase, we build the admin panel for the CMS Manager role. This includes:

- **Page Management**: Create and edit static pages (Home, About, Contact, Privacy Policy, Terms & Conditions) with a rich text editor.
- **Email Template Management**: View and edit HTML email templates for OTP, order confirmation, download links, etc., with a preview feature.
- **System Settings**: Manage key‑value settings (copyright year, contact email, site title, etc.).
- **Contact Messages**: View and manage messages submitted via the Contact page, with status tracking.

### Step 9.1: Update Prisma Schema

Add the following models to `prisma/schema.prisma`:

```prisma
model Page {
  id         String   @id @default(cuid())
  slug       String   @unique // e.g., "about", "privacy", "terms"
  title      String
  content    String   // HTML content
  published  Boolean  @default(true)
  createdAt  DateTime @default(now())
  updatedAt  DateTime @updatedAt
}

model EmailTemplate {
  id          String   @id @default(cuid())
  name        String   @unique // e.g., "welcome", "otp", "order-confirmation"
  subject     String
  bodyHtml    String   // HTML content with placeholders like {{name}}
  description String?
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt
}

model SystemSetting {
  id        String   @id @default(cuid())
  key       String   @unique
  value     String   // stored as string, parsed on use
  type      String   // "text", "number", "boolean", "email", etc.
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
}
```

Run the migration:

```bash
cd server
npx prisma migrate dev --name add_cms_models
```

```typescript
import { Request, Response } from 'express';
import { PrismaClient } from '@prisma/client';

const prisma = new PrismaClient();

// ========== Page Management ==========
export const getPages = async (req: Request, res: Response) => {
  try {
    const pages = await prisma.page.findMany({ orderBy: { slug: 'asc' } });
    res.json({ success: true, data: pages });
  } catch (error) {
    res.status(500).json({ success: false, error: 'Server error' });
  }
};

export const getPageBySlug = async (req: Request, res: Response) => {
  try {
    const { slug } = req.params;
    const page = await prisma.page.findUnique({ where: { slug } });
    if (!page) return res.status(404).json({ success: false, error: 'Page not found' });
    res.json({ success: true, data: page });
  } catch (error) {
    res.status(500).json({ success: false, error: 'Server error' });
  }
};

export const createPage = async (req: Request, res: Response) => {
  try {
    const { slug, title, content, published } = req.body;
    const existing = await prisma.page.findUnique({ where: { slug } });
    if (existing) return res.status(400).json({ success: false, error: 'Slug already exists' });
    const page = await prisma.page.create({ data: { slug, title, content, published } });
    res.status(201).json({ success: true, data: page });
  } catch (error) {
    res.status(500).json({ success: false, error: 'Server error' });
  }
};

export const updatePage = async (req: Request, res: Response) => {
  try {
    const { id } = req.params;
    const { title, content, published } = req.body;
    const page = await prisma.page.update({
      where: { id },
      data: { title, content, published },
    });
    res.json({ success: true, data: page });
  } catch (error) {
    res.status(500).json({ success: false, error: 'Server error' });
  }
};

export const deletePage = async (req: Request, res: Response) => {
  try {
    const { id } = req.params;
    await prisma.page.delete({ where: { id } });
    res.json({ success: true, message: 'Page deleted' });
  } catch (error) {
    res.status(500).json({ success: false, error: 'Server error' });
  }
};

// ========== Email Templates ==========
export const getEmailTemplates = async (req: Request, res: Response) => {
  try {

  } catch (error) {
    res.status(500).json({ success: false, error: 'Server error' });
  }
};

    if (!template) return res.status(404).json({ success: false, error: 'Template not found' });
    res.json({ success: true, data: template });
  } catch (error) {
    res.status(500).json({ success: false, error: 'Server error' });

  try {

    if (existing) return res.status(400).json({ success: false, error: 'Name already exists' });

    res.status(201).json({ success: true, data: template });

    res.json({ success: true, data: template });
  } catch (error) {
    res.status(500).json({ success: false, error: 'Server error' });

    const { id } = req.params;

    const setting = await prisma.systemSetting.findUnique({ where: { key } });
    if (!setting) return res.status(404).json({ success: false, error: 'Setting not found' });

  } catch (error) {
    res.status(500).json({ success: false, error: 'Server error' });

export const updateSystemSetting = async (req: Request, res: Response) => {
  try {
    const { key } = req.params;
    const { value } = req.body;
    const setting = await prisma.systemSetting.update({
      where: { key },
      data: { value },
    });
    res.json({ success: true, data: setting });
  } catch (error) {
    res.status(500).json({ success: false, error: 'Server error' });
  }
};

// ========== Contact Messages ==========
export const getContactMessages = async (req: Request, res: Response) => {
  try {
    const page = parseInt(req.query.page as string) || 1;
    const limit = parseInt(req.query.limit as string) || 20;
    const skip = (page - 1) * limit;
    const messages = await prisma.contactMessage.findMany({
      orderBy: { createdAt: 'desc' },
      skip,
      take: limit,
    });
    const total = await prisma.contactMessage.count();
    res.json({
      success: true,
      data: messages,
      pagination: { page, limit, total, pages: Math.ceil(total / limit) },
    });
  } catch (error) {
    res.status(500).json({ success: false, error: 'Server error' });
  }
};

export const updateContactMessageStatus = async (req: Request, res: Response) => {
  try {
    const { id } = req.params;
    const { status, adminNotes } = req.body;
    const message = await prisma.contactMessage.update({
      where: { id },
      data: { status, adminNotes },
    });
    res.json({ success: true, data: message });
  } catch (error) {
    res.status(500).json({ success: false, error: 'Server error' });
  }
};
```

### Step 9.4: Create CMS Routes

Create `server/src/routes/cms.routes.ts`:

```typescript
import { Router } from 'express';
import { protect } from '../middleware/auth.middleware';
import { authorize } from '../middleware/permission.middleware';
import {
  getPages,
  getPageBySlug,
  createPage,
  updatePage,
  deletePage,
  getEmailTemplates,
  getEmailTemplate,
  createEmailTemplate,
  updateEmailTemplate,
  deleteEmailTemplate,
  getSystemSettings,
  getSystemSetting,
  updateSystemSetting,
  getContactMessages,
  updateContactMessageStatus,
} from '../controllers/cms.controller';

const router = Router();

router.use(protect);
router.use(authorize('EDIT_PAGES', 'EDIT_EMAIL_TEMPLATES')); // at least one of these

// Pages
router.get('/pages', authorize('EDIT_PAGES'), getPages);
router.get('/pages/:slug', authorize('EDIT_PAGES'), getPageBySlug);
router.post('/pages', authorize('EDIT_PAGES'), createPage);
router.put('/pages/:id', authorize('EDIT_PAGES'), updatePage);
router.delete('/pages/:id', authorize('EDIT_PAGES'), deletePage);

// Email Templates
router.get('/email-templates', authorize('EDIT_EMAIL_TEMPLATES'), getEmailTemplates);
router.get('/email-templates/:id', authorize('EDIT_EMAIL_TEMPLATES'), getEmailTemplate);
router.post('/email-templates', authorize('EDIT_EMAIL_TEMPLATES'), createEmailTemplate);
router.put('/email-templates/:id', authorize('EDIT_EMAIL_TEMPLATES'), updateEmailTemplate);
router.delete('/email-templates/:id', authorize('EDIT_EMAIL_TEMPLATES'), deleteEmailTemplate);

// System Settings
router.get('/settings', authorize('EDIT_PAGES'), getSystemSettings); // reuse same permission
router.get('/settings/:key', authorize('EDIT_PAGES'), getSystemSetting);
router.put('/settings/:key', authorize('EDIT_PAGES'), updateSystemSetting);

// Contact Messages
router.get('/contact-messages', authorize('EDIT_PAGES'), getContactMessages);
router.put('/contact-messages/:id', authorize('EDIT_PAGES'), updateContactMessageStatus);

export default router;
```

Mount in `index.ts`:

```typescript

```

**`PagesList.tsx`**:

```tsx
import { useEffect, useState } from 'react';
import { Link } from 'react-router-dom';
import api from '../../services/api';
import toast from 'react-hot-toast';
import { PencilIcon, TrashIcon } from '@heroicons/react/24/outline';

interface Page {
  id: string;
  slug: string;
  title: string;
  published: boolean;
  updatedAt: string;
}

export default function PagesList() {
  const [pages, setPages] = useState<Page[]>([]);
  const [loading, setLoading] = useState(false);

  const fetchPages = async () => {
    setLoading(true);
    try {
      const res = await api.get('/admin/cms/pages');
      setPages(res.data.data);
    } catch (error) {
      toast.error('Failed to load pages');
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    fetchPages();
  }, []);

  const handleDelete = async (id: string) => {
    if (!confirm('Delete this page?')) return;
    try {
      await api.delete(`/admin/cms/pages/${id}`);
      toast.success('Page deleted');
      fetchPages();
    } catch (error) {
      toast.error('Failed to delete');
    }
  };

  return (
    <div>
      <div className="flex justify-between items-center mb-6">
        <h1 className="text-2xl font-bold text-[#723E31]">Pages</h1>
        <Link to="/admin/cms/pages/new" className="bg-[#A67564] text-white px-4 py-2 rounded hover:bg-[#723E31]">
          + New Page
        </Link>
      </div>
      {loading && <div>Loading...</div>}
      <div className="bg-white shadow overflow-hidden sm:rounded-md">
        <ul className="divide-y divide-gray-200">
          {pages.map(page => (
            <li key={page.id} className="px-6 py-4 flex items-center justify-between">
              <div>
                <p className="font-medium">{page.title}</p>
                <p className="text-sm text-gray-500">Slug: {page.slug}</p>
                <p className="text-xs text-gray-400">Updated: {new Date(page.updatedAt).toLocaleDateString()}</p>
              </div>
              <div className="flex space-x-2">
                <Link to={`/admin/cms/pages/${page.id}/edit`} className="text-[#A67564] hover:text-[#723E31]">
                  <PencilIcon className="h-5 w-5" />
                </Link>
                <button onClick={() => handleDelete(page.id)} className="text-red-600 hover:text-red-800">
                  <TrashIcon className="h-5 w-5" />
                </button>
              </div>
            </li>
          ))}
        </ul>
      </div>
    </div>
  );
}
```

**`PageForm.tsx`** (using simple textarea):

```tsx
import { useEffect, useState } from 'react';
import { useForm } from 'react-hook-form';
import { useNavigate, useParams } from 'react-router-dom';
import api from '../../services/api';
import toast from 'react-hot-toast';

interface FormData {
  slug: string;
  title: string;
  content: string;
  published: boolean;
}

export default function PageForm() {
  const { id } = useParams<{ id: string }>();
  const navigate = useNavigate();
  const [loading, setLoading] = useState(false);
  const { register, handleSubmit, setValue, watch } = useForm<FormData>();

  useEffect(() => {
    if (id) {
      setLoading(true);
      api.get(`/admin/cms/pages/${id}`) // actually need to fetch by id, but our route uses slug. We'll need an id endpoint or fetch all and find.
        .then(res => {
          const page = res.data.data;
          setValue('slug', page.slug);
          setValue('title', page.title);
          setValue('content', page.content);
          setValue('published', page.published);
        })
        .catch(() => toast.error('Failed to load page'))
        .finally(() => setLoading(false));
    }
  }, [id, setValue]);

  const onSubmit = async (data: FormData) => {
    try {
      if (id) {
        await api.put(`/admin/cms/pages/${id}`, data);
        toast.success('Page updated');
      } else {
        await api.post('/admin/cms/pages', data);
        toast.success('Page created');
      }
      navigate('/admin/cms/pages');
    } catch (error: any) {
      toast.error(error.response?.data?.error || 'Operation failed');
    }
  };

  const content = watch('content');

  return (
    <div className="max-w-4xl mx-auto bg-white p-6 rounded shadow">
      <h1 className="text-2xl font-bold text-[#723E31] mb-6">{id ? 'Edit Page' : 'New Page'}</h1>
      <form onSubmit={handleSubmit(onSubmit)} className="space-y-4">
        <div>
          <label className="block text-sm font-medium mb-1">Slug *</label>
          <input {...register('slug', { required: true })} className="w-full border rounded px-3 py-2" disabled={!!id} />
        </div>
        <div>
          <label className="block text-sm font-medium mb-1">Title *</label>
          <input {...register('title', { required: true })} className="w-full border rounded px-3 py-2" />
        </div>
        <div>
          <label className="block text-sm font-medium mb-1">Content (HTML) *</label>
          <textarea {...register('content', { required: true })} rows={10} className="w-full border rounded px-3 py-2 font-mono" />
        </div>
        <div className="flex items-center">
          <input {...register('published')} type="checkbox" id="published" className="mr-2" />
          <label htmlFor="published">Published</label>
        </div>
        <div>
          <p className="text-sm font-medium mb-1">Preview:</p>
          <div className="border p-4 rounded bg-gray-50" dangerouslySetInnerHTML={{ __html: content }} />
        </div>
        <div className="flex justify-end space-x-3">
          <button type="button" onClick={() => navigate('/admin/cms/pages')} className="px-4 py-2 border rounded">Cancel</button>
          <button type="submit" className="px-4 py-2 bg-[#A67564] text-white rounded">{id ? 'Update' : 'Create'}</button>
        </div>
      </form>
    </div>
  );
}
```
